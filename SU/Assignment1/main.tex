\section{Examples 1}
\subsection{Question 1}

Find the distance between the following pairs of points
\begin{enumerate}
\begin{align}
\myvec{1\\-3}, \myvec{-2\\1}
\end{align}
\end{enumerate}
\end{enumerate}
\subsection{Solution}

\item The distance between two vectors is given by 
\begin{align}
\norm{\vec{A}-\vec{B}}
\label{eq5}
\end{align}
From \eqref{eq5}
\begin{align}
\norm{\vec{A}-\vec{B}}&= \sqrt{\myvec{\vec{A}-\vec{B}}^\intercal\myvec{\vec{A}-\vec{B}}}\\
&=\norm{\myvec{1\\-3}-\myvec{-2\\1}} \\
&=\norm{\myvec{3\\-4}} \\
&=\sqrt{\myvec{3& -4}& \myvec{3\\-4}} \\
&=\sqrt{3^2& \myvec{-4}^2} \\
&=\sqrt{\myvec{9+16}}  \nonumber \\
&=5 \nonumber  
\end{align}
