\section{Examples 1}
\subsection{Question 1}

Find the area of the triangle of coordinates whose angular points are (1,3), (-7,6), (5,-1).
\end{enumerate}
\subsection{Solution}
\begin{align}
& \frac{1}{2}\mydet{\myvec{A-B}\myvec{A-C}}
\label{eq5}
\end{align}
Let $\vec{A}=\myvec{1\\3}, \vec{B}=\myvec{-7\\6}, \vec{C}=\myvec{5\\-1}\\

\begin{align}
\vec{A}-\vec{B}&=\myvec{1\\3}-\myvec{-7\\6}\\
&=\myvec{8\\-3}\\
\vec{A}-\vec{C}&=\myvec{1\\3}-\myvec{5\\-1}\\
&=\myvec{-4\\4}
\end{align}
From \eqref{eq5}
\item Area of the triangle
\begin{align}
&=\frac{1}{2}\mydet{8&-4\\-3&4}\\
&=\frac{1}{2}\left[\myvec{8\times4}-\myvec{-3\times-4}\right]\\
&=\frac{1}{2}\myvec{32-12}\\
&=\frac{1}{2}\myvec{20}\\
&=10\nonumber
\end{align}
